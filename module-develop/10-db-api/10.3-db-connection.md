$query = $connection->select('users', 'u');
// where条件
$query->condition('u.uid', 0, '<>');
// 查询字段
$query->fields('u', ['uid', 'name', 'status', 'created', 'access']);
// 查询范围
$query->range(0, 50);
// 执行查询 构建查询后，调用execute()方法来编译并运行查询
$query->execute();
// 循环方式输出

}
